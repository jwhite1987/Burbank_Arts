def number_DivIcon(color):
    """ Create a 'numbered' icon
    
    """
    icon = DivIcon(
            icon_size=(0,0),
            icon_anchor=(0,0),
            html="""<span class="dot" style="background-color : {:s}; height: 7px; width: 7px; border-radius: 50%; display: inline-block;"></span>
                """.format(color)
        )
    return icon
